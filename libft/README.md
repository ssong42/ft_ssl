# NAME

  ft_printf -- formatted output
 
# SYNOPSIS

  ft_printf(const char *str, arguments ...)

# Range
Can Handle the format identifiers
  d, i, o, u, X, x, s, S, c, C, %

Can Handle the flags
  l, ll, h, hh, j, z
